import json


def convert_json_booleans(json_str):
    def convert_values(obj):
        if isinstance(obj, dict):
            return {k: convert_values(v) for k, v in obj.items()}
        elif isinstance(obj, list):
            return [convert_values(i) for i in obj]
        elif isinstance(obj, str) and obj.lower() == 'true':
            return True
        elif isinstance(obj, str) and obj.lower() == 'false':
            return False
        return obj

    # Parse the JSON string to a Python object
    data = json.loads(json_str)

    # Convert the values
    converted_data = convert_values(data)

    return converted_data


# Example usage
json_input = '{"apple": false}'
result = convert_json_booleans(json_input)
print(result)  # Output: {'apple': True}
